local function ResolveIcon(name, icon)
	if icon == "rbxassetid://0" or icon == "" or icon == nil then
		return DefaultIcons[name] or "rbxassetid://0"
	end
	return icon
end

local function ResolveFont(base)
	if Config.CustomFont and Config.CustomFont ~= "" then
		local ok, f = pcall(function()
			return Font.new(Config.CustomFont, Enum.FontWeight.Regular)
		end)
		if ok and f then return f end
	end
	return base
end

local function RF(base)  return ResolveFont(base) end
local function RFB(base) return ResolveFont(base) end

local function ResolveTextColor()
	if type(Config.Text_Color) == "userdata" then return Config.Text_Color end
	local r, g, b   = Config.Color.R, Config.Color.G, Config.Color.B
	local luminance  = 0.2126 * r + 0.7152 * g + 0.0722 * b
	return luminance > 0.55 and Color3.fromRGB(0, 0, 0) or Color3.fromRGB(255, 255, 255)
end

local function GetFont(weight)
	if Config.CustomFont and Config.CustomFont ~= "" then
		local w = weight == "bold"  and Enum.FontWeight.Bold
		       or weight == "black" and Enum.FontWeight.ExtraBold
		       or Enum.FontWeight.Regular
		local ok, f = pcall(Font.new, Config.CustomFont, w)
		if ok then return f end
	end
	if weight == "bold"  then return Enum.Font.GothamBold  end
	if weight == "black" then return Enum.Font.GothamBlack end
	return Enum.Font.Gotham
end

local function ApplyFont(lbl, weight)
	local f = GetFont(weight)
	if typeof(f) == "Font" then
		lbl.FontFace = f
	else
		lbl.Font = f
	end
end

local function GetVisibleProviders()
	if Config.UseJunkie then
		for _, p in ipairs(Config.Providers) do
			if p.IsJunkie and p.Show ~= false then
				return { p }
			end
		end
	end
	local pinned, rest = {}, {}
	for _, p in ipairs(Config.Providers) do
		if p.Show ~= false and not p.IsJunkie then
			if p.Pinned then
				table.insert(pinned, p)
			else
				table.insert(rest, p)
			end
		end
	end
	local out = {}
	for _, p in ipairs(pinned) do table.insert(out, p) end
	for _, p in ipairs(rest)   do table.insert(out, p) end
	return out
end

local function FmtNum(n)
	local s = tostring(math.floor(n))
	local r, l = "", #s
	for i = 1, l do
		if i > 1 and (l - i + 1) % 3 == 0 then r = r .. "," end
		r = r .. s:sub(i, i)
	end
	return r
end

local function MakeDraggable(frame, hitbox)
	frame.AnchorPoint = Vector2.new(0, 0)
	local dragging, dragConn    = false, nil
	local startMouse, startPos  = Vector2.new(), Vector2.new()

	local function getInputPos(inp)
		return Vector2.new(inp.Position.X, inp.Position.Y)
	end

	hitbox.InputBegan:Connect(function(inp)
		if inp.UserInputType ~= Enum.UserInputType.MouseButton1
		and inp.UserInputType ~= Enum.UserInputType.Touch then return end
		dragging   = true
		startMouse = getInputPos(inp)
		startPos   = Vector2.new(frame.AbsolutePosition.X, frame.AbsolutePosition.Y)
		if dragConn then dragConn:Disconnect() end
		dragConn = RunService.RenderStepped:Connect(function()
			if not dragging then dragConn:Disconnect(); return end
			local cur = UserInputService:GetMouseLocation()
			local s   = SS()
			local fw  = frame.AbsoluteSize.X
			local fh  = frame.AbsoluteSize.Y
			local nx  = math.clamp(startPos.X + (cur.X - startMouse.X), 0, math.max(0, s.X - fw))
			local ny  = math.clamp(startPos.Y + (cur.Y - startMouse.Y), 0, math.max(0, s.Y - fh))
			frame.Position = UDim2.fromOffset(nx, ny)
		end)
	end)

	UserInputService.InputEnded:Connect(function(inp)
		if inp.UserInputType == Enum.UserInputType.MouseButton1
		or inp.UserInputType == Enum.UserInputType.Touch then
			dragging = false
		end
	end)
end

local function GenerateToken()
	if crypt and crypt.hash then
		return crypt.hash(tick() .. tostring(math.random()), "sha256")
	end
	return tostring(tick()) .. tostring(math.random()) .. tostring(math.random())
end

function Hub:FetchDiscord()
	if not self.cfg.ShowMemberCount then
		if self.memberLbl then self.memberLbl.Parent.Visible = false end
		return
	end
	task.spawn(function()
		local HS  = game:GetService("HttpService")
		local req = (syn and syn.request) or (http and http.request) or http_request or (fluxus and fluxus.request) or request
		if not req then
			if self.memberLbl then self.memberLbl.Text = "" end
			return
		end
		local ok, res = pcall(req, {
			Url     = "https://discord.com/api/v10/invites/" .. self.cfg.Discord .. "?with_counts=true",
			Method  = "GET",
			Headers = { ["Content-Type"] = "application/json" },
		})
		if ok and res and res.StatusCode == 200 then
			local d = HS:JSONDecode(res.Body)
			if d and d.approximate_member_count and self.memberLbl then
				local r = math.floor(self.cfg.Color.R * 255)
				local g = math.floor(self.cfg.Color.G * 255)
				local b = math.floor(self.cfg.Color.B * 255)
				self.memberLbl.Text = string.format(
					"<font color='rgb(%d,%d,%d)'>%s</font> members  •  <font color='rgb(70,200,70)'>%s online</font>",
					r, g, b,
					FmtNum(d.approximate_member_count),
					FmtNum(d.approximate_presence_count or 0)
				)
			end
		else
			if self.memberLbl then self.memberLbl.Text = "" end
		end
	end)
end

function Hub:SpawnSnowflake(sg, ov)
	local s        = SS()
	local snowSz   = self.cfg.SnowSize or { Min = 3, Max = 7 }
	local sz       = math.random(snowSz.Min, snowSz.Max)
	local x        = math.random(0, s.X)
	local spd      = math.random(60, 140)
	local drift    = math.random(-18, 18)
	local rot      = math.random(0, 360)
	local rotSpd   = math.random(-90, 90)
	local opacity  = math.random(55, 90) / 100
	local flakeColor = self.cfg.SnowColor or Color3.fromRGB(220, 235, 255)

	local flake = Instance.new("Frame", ov)
	flake.Size                  = UDim2.fromOffset(sz, sz)
	flake.Position              = UDim2.fromOffset(x, -sz)
	flake.BackgroundColor3      = flakeColor
	flake.BackgroundTransparency = 1 - opacity
	flake.BorderSizePixel       = 0
	flake.ZIndex                = 2
	flake.Rotation              = rot
	Instance.new("UICorner", flake).CornerRadius = UDim.new(0.5, 0)

	local startTime = tick()
	local startX    = x
	local conn
	conn = RunService.RenderStepped:Connect(function()
		if not flake or not flake.Parent then
			conn:Disconnect(); return
		end
		local elapsed = tick() - startTime
		local ny      = -sz + elapsed * spd
		local nx      = startX + math.sin(elapsed * 0.9) * drift
		flake.Position = UDim2.fromOffset(nx, ny)
		flake.Rotation = rot + elapsed * rotSpd
		if ny > s.Y + sz then
			conn:Disconnect()
			flake:Destroy()
		end
	end)
end

function Hub:StartSnow(sg, ov)
	if not self.cfg.SnowEnabled then return end
	local density      = math.max(0.1, self.cfg.SnowDensity or 1.0)
	local spawnInterval = 0.13 / density

	local function spawnBatch()
		local count = math.max(1, math.floor(3 * density))
		for _ = 1, count do
			self:SpawnSnowflake(sg, ov)
		end
	end

	spawnBatch()
	task.spawn(function()
		while sg and sg.Parent do
			task.wait(math.random(math.floor(spawnInterval * 60), math.floor(spawnInterval * 140)) / 100)
			if sg and sg.Parent then
				self:SpawnSnowflake(sg, ov)
			end
		end
	end)
end

function Hub:SwitchMainTab(tab)
	self.activeMainTab = tab
	local isIntro     = tab == "intro"
	local isChangelog = tab == "changelog"
	local isKey       = tab == "key"
	if self.introView     then self.introView.Visible     = isIntro     end
	if self.keyView       then self.keyView.Visible       = isKey       end
	if self.changelogView then self.changelogView.Visible = isChangelog end
	if self.toggleBtnLbl  then
		self.toggleBtnLbl.Text = (isIntro or isChangelog) and "←" or "?"
	end
end

function Hub:TogglePanel()
	local p = self.sidePanel
	if not p then return end
	self.panelOpen = not self.panelOpen
	if self.panelOpen then
		local bgP = self.bg.AbsolutePosition
		local bgS = self.bg.AbsoluteSize
		local pW  = p.AbsoluteSize.X
		local pH  = p.AbsoluteSize.Y
		local s   = SS()
		local px  = bgP.X + bgS.X + 8
		if px + pW > s.X - 4 then px = bgP.X - pW - 8 end
		if px < 4 then px = math.max(4, bgP.X + bgS.X - pW - 2) end
		local py = bgP.Y
		if py + pH > s.Y - 4 then py = math.max(4, s.Y - pH - 4) end
		p.AnchorPoint           = Vector2.new(0, 0)
		p.Position              = UDim2.fromOffset(px, py)
		p.BackgroundTransparency = 1
		p.Visible               = true
		TweenService:Create(p, TweenInfo.new(0.2, Enum.EasingStyle.Back, Enum.EasingDirection.Out), {
			BackgroundTransparency = 0
		}):Play()
	else
		TweenService:Create(p, TweenInfo.new(0.15), { BackgroundTransparency = 1 }):Play()
		task.delay(0.17, function() if p then p.Visible = false end end)
	end
end

function Hub:Shake()
	local bg = self.bg
	if not bg then return end
	local px = bg.AbsolutePosition.X
	local py = bg.AbsolutePosition.Y
	local s  = SS()
	local fw = bg.AbsoluteSize.X
	local fh = bg.AbsoluteSize.Y
	px = math.clamp(px, 0, math.max(0, s.X - fw))
	py = math.clamp(py, 0, math.max(0, s.Y - fh))
	task.spawn(function()
		for _, o in ipairs({ 8, -8, 6, -6, 4, -4, 2, -2, 0 }) do
			local nx = math.clamp(px + o, 0, math.max(0, s.X - fw))
			bg.Position = UDim2.fromOffset(nx, py)
			task.wait(0.03)
		end
		bg.Position = UDim2.fromOffset(px, py)
	end)
end

function Hub:Redeem()
	if self.keyInput == "" then
		self:Shake()
		self:Toast("Enter a key first.", "error")
		return
	end

	local key    = self.keyInput:upper()
	self.keyInput = key
	if self.textBox then self.textBox.Text = key end

	self.keyAttempts = self.keyAttempts + 1
	if self.keyAttempts > self.cfg.MaxKeyAttempts then
		self:Toast("Too many attempts.", "error")
		task.delay(1, function() self:Close() end)
		return
	end

	local remaining = self.cfg.MaxKeyAttempts - self.keyAttempts
	if self.attemptsLbl then
		self.attemptsLbl.Text = tostring(remaining) .. " attempt" .. (remaining == 1 and "" or "s") .. " remaining"
		self.attemptsLbl.TextColor3 = remaining <= 1 and Color3.fromRGB(220, 55, 55)
		                           or remaining <= 2 and Color3.fromRGB(230, 160, 30)
		                           or Color3.fromRGB(255, 255, 255)
	end

	if self.cfg.UseJunkie and not self.junkie then
		self:Shake()
		self:Toast("SDK still loading — try again.", "error")
		self.keyAttempts = self.keyAttempts - 1
		return
	end

	if self.junkie then
		local result = self.junkie.check_key(key)
		if result and result.valid then
			self.validatedToken = self.authToken
			local keyToStore    = (result.message == "KEYLESS") and "KEYLESS" or key
			getgenv().SCRIPT_KEY = keyToStore
			pcall(function() if writefile then writefile(self.keyFile, keyToStore) end end)
			self:Toast(self.cfg.SuccessMessage, "success")
			self:SendWebhook(key)
			if self.cfg.CloseOnSuccess then
				task.delay(0.8, function() self:LoadScript() end)
			end
		else
			local err = result and result.error or "Unknown error"
			self:Shake()
			if     err == "KEY_EXPIRED"       then self:Toast("Key expired — get a new one.", "error")
			elseif err == "HWID_BANNED"       then self:Toast("Hardware banned.", "error"); task.delay(1, function() Players.LocalPlayer:Kick("Hardware banned.") end)
			elseif err == "KEY_INVALIDATED"   then self:Toast("Key was disabled.", "error")
			elseif err == "ALREADY_USED"      then self:Toast("One-time key already used.", "error")
			elseif err == "HWID_MISMATCH"     then self:Toast("HWID limit reached.", "error")
			elseif err == "PREMIUM_REQUIRED"  then self:Toast("Premium key required.", "error")
			elseif err == "SERVICE_MISMATCH"  then self:Toast("Key is for a different service.", "error")
			elseif err == "SERVICE_NOT_FOUND" then self:Toast("Service not found — check your config.", "error")
			elseif err == "KEY_INVALID"       then self:Toast("Invalid key.", "error")
			elseif err == "ERROR"             then self:Toast("Network error — try again.", "error")
			elseif tostring(err):match("^http %d+") then self:Toast("Server error: " .. tostring(err), "error")
			else self:Toast("✗ " .. tostring(err), "error") end
		end
	else
		self.validatedToken  = self.authToken
		getgenv().SCRIPT_KEY = key
		pcall(function() if writefile then writefile(self.keyFile, key) end end)
		self:Toast(self.cfg.SuccessMessage, "success")
		self:SendWebhook(key)
		if self.cfg.CloseOnSuccess then
			task.delay(0.8, function() self:LoadScript() end)
		end
	end
end

			}),
		})
	end

		TweenService:Create(self.blurFx, TweenInfo.new(0.4), { Size = 0 }):Play()
		task.delay(0.4, function() self.blurFx:Destroy() end)

	if self.gui then self.gui:Destroy() end

				local bullet2 = Instance.new("Frame", noteRow)
				bullet2.Size                  = UDim2.fromOffset(4, 4)
				bullet2.BackgroundColor3      = isLatest and tagCol or Color3.fromRGB(70, 75, 105)
				bullet2.BackgroundTransparency = 0.1
				bullet2.BorderSizePixel       = 0
				MkCorner(bullet2, 99)
				local bp2 = Instance.new("UIPadding", bullet2)
				bp2.PaddingTop = UDim.new(0, 4)

				local ntxt2 = Instance.new("TextLabel", noteRow)
				ntxt2.Size                  = UDim2.new(1, -10, 0, 0)
				ntxt2.AutomaticSize         = Enum.AutomaticSize.Y
				ntxt2.BackgroundTransparency = 1
				ntxt2.Text                  = noteStr
				ntxt2.TextColor3            = isLatest and Color3.fromRGB(200, 205, 228) or Color3.fromRGB(148, 153, 178)
				ntxt2.TextSize              = 10
				ntxt2.TextXAlignment        = Enum.TextXAlignment.Left
				ntxt2.TextWrapped           = true
				ApplyFont(ntxt2, "regular")
			end
		end
	else
		-- No changelog configured, show placeholder
		local noCl = Instance.new("TextLabel", rightCol)
		noCl.Size                  = UDim2.new(1, 0, 0, 60)
		noCl.BackgroundTransparency = 1
		noCl.Text                  = "No changelog entries yet."
		noCl.TextColor3            = Color3.fromRGB(80, 85, 115)
		noCl.TextSize              = 11
		noCl.TextXAlignment        = Enum.TextXAlignment.Center
		noCl.TextYAlignment        = Enum.TextYAlignment.Center
		ApplyFont(noCl, "regular")
	end

	-- Socials strip at bottom of center col
	local visS = GetVisibleSocials()
	if #visS > 0 then
		local socialH2 = 30
		local socialScroll = Instance.new("ScrollingFrame", centerCol)
		socialScroll.Size                 = UDim2.new(1, 0, 0, socialH2 + 4)
		socialScroll.BackgroundTransparency = 1
		socialScroll.BorderSizePixel      = 0
		socialScroll.ScrollBarThickness   = 0
		socialScroll.ScrollingDirection   = Enum.ScrollingDirection.X
		socialScroll.CanvasSize           = UDim2.fromOffset(0, 0)
		socialScroll.AutomaticCanvasSize  = Enum.AutomaticSize.X
		socialScroll.ElasticBehavior      = Enum.ElasticBehavior.WhenScrollable

		local sstrip = Instance.new("Frame", socialScroll)
		sstrip.Size                  = UDim2.new(0, 0, 1, 0)
		sstrip.AutomaticSize         = Enum.AutomaticSize.X
		sstrip.BackgroundTransparency = 1
		local ssl = Instance.new("UIListLayout", sstrip)
		ssl.FillDirection       = Enum.FillDirection.Horizontal
		ssl.VerticalAlignment   = Enum.VerticalAlignment.Center
		ssl.Padding             = UDim.new(0, 4)

		local chipW2, chipH2 = 72, socialH2
		for _, sd in ipairs(visS) do
			local chip2 = Instance.new("Frame", sstrip)
			chip2.Size                  = UDim2.fromOffset(chipW2, chipH2)
			chip2.BackgroundColor3      = Color3.fromRGB(14, 18, 32)
			chip2.BorderSizePixel       = 0
			MkCorner(chip2, 99)
			MkStroke(chip2, self.cfg.Color, sd.Pinned and 1.5 or 1, sd.Pinned and 0.1 or 0.65)

			local cg2 = Instance.new("UIGradient", chip2)
			cg2.Color = ColorSequence.new({
				ColorSequenceKeypoint.new(0, self.cfg.Color:Lerp(Color3.fromRGB(10, 14, 28), 0.8)),
				ColorSequenceKeypoint.new(1, Color3.fromRGB(10, 14, 28)),
			})

			local inf2 = Instance.new("Frame", chip2)
			inf2.Size                  = UDim2.new(1, 0, 1, 0)
			inf2.BackgroundTransparency = 1
			local il2 = Instance.new("UIListLayout", inf2)
			il2.Padding             = UDim.new(0, 4)
			il2.FillDirection       = Enum.FillDirection.Horizontal
			il2.HorizontalAlignment = Enum.HorizontalAlignment.Center
			il2.VerticalAlignment   = Enum.VerticalAlignment.Center
			local ip2 = Instance.new("UIPadding", inf2)
			ip2.PaddingLeft = UDim.new(0, 6); ip2.PaddingRight = UDim.new(0, 6)

			local si2 = Instance.new("ImageLabel", inf2)
			si2.Size                  = UDim2.fromOffset(12, 12)
			si2.BackgroundTransparency = 1
			si2.Image                 = ResolveIcon(sd.Name, sd.Icon)
			si2.ImageColor3           = Color3.fromRGB(255, 255, 255)

			local st2 = Instance.new("TextLabel", inf2)
			st2.Size                  = UDim2.new(0, 0, 1, 0)
			st2.AutomaticSize         = Enum.AutomaticSize.X
			st2.BackgroundTransparency = 1
			st2.Text                  = sd.Name
			st2.TextColor3            = Color3.fromRGB(255, 255, 255)
			st2.TextTransparency      = 0.1
			st2.TextSize              = 9
			st2.ClipsDescendants      = true
			ApplyFont(st2, "bold")

			local sc2 = Instance.new("TextButton", chip2)
			sc2.Size                  = UDim2.new(1, 0, 1, 0)
			sc2.BackgroundTransparency = 1
			sc2.Text                  = ""
			sc2.ZIndex                = 10
			local lnk2, nm2 = sd.Link, sd.Name
			sc2.MouseButton1Click:Connect(function()
				if setclipboard then setclipboard(lnk2) end
				self:Toast(nm2 .. " copied!", "success")
			end)
			sc2.MouseEnter:Connect(function()
				TweenService:Create(chip2, TweenInfo.new(0.12), { BackgroundColor3 = self.cfg.Color:Lerp(Color3.fromRGB(8, 12, 22), 0.5) }):Play()
			end)
			sc2.MouseLeave:Connect(function()
				TweenService:Create(chip2, TweenInfo.new(0.12), { BackgroundColor3 = Color3.fromRGB(14, 18, 32) }):Play()
			end)
		end
	end

	-- Still build the side panel for socials popup (triggered via toggle button)
	self:BuildSidePanel()
end

return Hub
