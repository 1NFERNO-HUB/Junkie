local function _hasBadPath(s)
	if not s then return false end
	local sl=_fn.tostring(s):lower()
	for _,p in _fn.ipairs(_badPaths) do
		if sl:find(p:lower(),1,true) then return true end
	end
	return false
end

local function _flag(reason)
	if _dead then return end
	_dead=true
	_fn.tins(_log,"flag="..(_fn.tostring(reason)))
	_fn.tins(_log,"clock=".._fn.tostring(_fn.osclock()))
	_fn.tins(_log,"thread=".._fn.tostring(_fn.crun()))
	_fn.pcall(_fn.cclose,_mt)
	coroutine.yield()
end

local function _try(f,label)
	local ok,err=_fn.pcall(f)
	if not ok then
		if _hasBadPath(err) then _flag(label.."|errpath") end
	end
end

local function _stackProbe(err,label)
	if _hasBadPath(err) then
		_flag(label.."|errpath") return
	end
	for i=0,20 do
		local ok,fe=_fn.pcall(_fn.getfenv,i)
		if ok and fe and fe~=_env then
			local keys={}
			for k in _fn.pairs(fe) do _fn.tins(keys,_fn.tostring(k)) end
			if #keys>0 then
				_flag(label.."|stackleak@"..i..":"..(_fn.tcat(keys,",")))
			end
		end
	end
end

local function _guiDestroy(g)
	_fn.pcall(function() g:Destroy() end)
end

local function _isSpyName(name)
	local nl=name:lower()
	for _,s in _fn.ipairs(_spyNameSubs) do
		if nl:find(s,1,true) then return true end
	end
	return false
end

local function _isSpyUrl(url)
	if _fn.type(url)~="string" then return false end
	local ul=url:lower()
	for _,d in _fn.ipairs(_spyDomains) do
		if ul:find(d,1,true) then return true end
	end
	return false
end

local function _scanLabels(root)
	_fn.pcall(function()
		for _,d in _fn.ipairs(root:GetDescendants()) do
			_fn.pcall(function()
				if d:IsA("TextLabel") and _spyTextExact[d.Text] then
					local t=d
					while t and not t:IsA("ScreenGui") do t=t.Parent end
					if t and t:IsA("ScreenGui") then _guiDestroy(t) end
				end
			end)
		end
	end)
end

local function _killNamedIn(parent)
	if not parent then return end
	_fn.pcall(function()
		for _,c in _fn.ipairs(parent:GetChildren()) do
			_fn.pcall(function()
				if (c:IsA("ScreenGui") or c:IsA("Frame")) and _spyGuiExact[c.Name] then
					_guiDestroy(c)
				end
			end)
		end
	end)
end

local function _heuristic(gui)
	_fn.pcall(function()
		if _knownGuis[gui] then return end
		if _guiWl[gui.Name] then return end
		if not _isSpyName(gui.Name) then return end
		local tl,sb=0,0
		for _,d in _fn.ipairs(gui:GetDescendants()) do
			if d:IsA("TextLabel") then tl=tl+1 end
			if d:IsA("ScrollingFrame") then sb=sb+1 end
		end
		if sb>=3 and tl>=50 then _guiDestroy(gui) end
	end)
end

local function _killWindUI()
	_fn.pcall(function()
		local re=_cg["54b3b10efc668b8054b210e563c00f41c67160da77a903ea2a16c3276824be7a"]
		if not re then return end
		local rn=re:FindFirstChild("WindUI")
		if not rn then return end
		local r0=rn:FindFirstChild("Window")
		if not r0 then return end
		local ch=r0:GetChildren()
		if ch[2] then _scanLabels(ch[2]) end
	end)
end

local function _wipeSpyGlobals()
	local rg=getgenv and getgenv()
	for _,k in _fn.ipairs(_spyGlobals) do
		if rg then _fn.pcall(function()
			if _fn.rawget(rg,k)~=nil then _fn.rawset(rg,k,nil) end
		end) end
		_fn.pcall(function()
			if _fn.rawget(_G,k)~=nil then _fn.rawset(_G,k,nil) end
		end)
	end
end

local function _scanParent(parent,heuristic)
	if not parent then return end
	_fn.pcall(function()
		for _,c in _fn.ipairs(parent:GetChildren()) do
			_fn.pcall(function()
				if c:IsA("ScreenGui") and not _knownGuis[c] then
					task.delay(0.05,function()
						if c and c.Parent then
							if heuristic then _heuristic(c) end
							_scanLabels(c)
						end
					end)
				end
			end)
		end
	end)
end

local function _snapshot()
	_fn.pcall(function()
		for _,c in _fn.ipairs(_cg:GetChildren()) do _knownGuis[c]=true end
	end)
	_fn.pcall(function()
		if _lp then
			local pg=_lp:FindFirstChildOfClass("PlayerGui")
			if pg then for _,c in _fn.ipairs(pg:GetChildren()) do _knownGuis[c]=true end end
		end
	end)
	_fn.pcall(function()
		local hui=gethui and gethui()
		if hui then for _,c in _fn.ipairs(hui:GetChildren()) do _knownGuis[c]=true end end
	end)
end

local function _sweep()
	_fn.pcall(function()
		if _lp then
			local pg=_lp:FindFirstChildOfClass("PlayerGui")
			if pg then _killNamedIn(pg); _scanParent(pg,false) end
		end
	end)
	_fn.pcall(function() _killNamedIn(_cg); _scanParent(_cg,true) end)
	_fn.pcall(function()
		local hui=gethui and gethui()
		if hui then _killNamedIn(hui); _scanParent(hui,true) end
	end)
	_killWindUI()
	_wipeSpyGlobals()
end

local function _onPlayerGui(c)
	task.defer(function()
		if not c or not c.Parent then return end
		if _spyGuiExact[c.Name] then _guiDestroy(c); return end
		_fn.pcall(function() _scanLabels(c) end)
	end)
end

local function _onCoreGui(c)
	task.defer(function()
		if not c or not c.Parent then return end
		if _spyGuiExact[c.Name] then _guiDestroy(c); return end
		_fn.pcall(function() _scanLabels(c) end)
		_fn.pcall(function()
			task.delay(0.15,function()
				if c and c.Parent then _heuristic(c) end
			end)
		end)
	end)
end

local function _hookHttp()
	_fn.pcall(function()
		local mt=getrawmetatable and getrawmetatable(_hs)
		if not mt then return end
		local orig=mt.__index
		setreadonly(mt,false)
		mt.__index=newcclosure(function(self,key)
			local fn=orig(self,key)
			if key=="RequestAsync" or key=="GetAsync" or key=="PostAsync" then
				return newcclosure(function(...)
					local args={...}
					local url=args[1]
					if _fn.type(url)=="table" then url=url.Url end
					if _isSpyUrl(url) then
						return {Success=false,StatusCode=403,Body="",Headers={}}
					end
					return fn(self,_unpack(args))
				end)
			end
			return fn
		end)
		setreadonly(mt,true)
	end)
end

	end)
	_fn.pcall(function()

				if _isSpyUrl(opts and opts.Url) then return {StatusCode=403,Body=""} end
				return o(opts)
			end)
		end
	end)
	_fn.pcall(function()
		if _fn.type(request)=="function" then
			local o=request
			request=newcclosure(function(opts)
				if _isSpyUrl(opts and opts.Url) then return {StatusCode=403,Body=""} end
				return o(opts)
			end)
		end
	end)
end

local function _hookRequire()
	_fn.pcall(function()
		local orig=_fn.require
		local h=newcclosure(function(mod)
			local ok,res=_fn.pcall(orig,mod)
			if not ok then return nil end
			if _fn.type(res)=="table" then
				for k in _fn.pairs(res) do
					local ks=_fn.tostring(k):lower()
					if ks:find("spy",1,true) or ks:find("hydroxide",1,true)
						or ks:find("exotool",1,true) then
						return nil
					end
				end
			end
			return res
		end)
		require=h
		_fn.require=h
	end)
end
